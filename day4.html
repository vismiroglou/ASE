Optionally, you can add the time series and Welch PSD/PS estimates as well, allow for dynamic updates of all figures when zooming, etc.. def generate_signal(N, fs, K, plot=False):
    """
    Generate a dynamic signal as a sum of K chirps with random amplitudes and frequencies.

    Parameters
    ----------
    N : int
        Length of the signal.
    fs : int or float
        Sampling frequency.
    K : int
        Number of chirps (orders) to sum.
    plot : bool, optional
        If True, plot the generated signal.

    Returns
    -------
    x : np.ndarray
        Generated signal.
    fs : int or float
        Sampling frequency (echoed).
    f : np.ndarray
        Array of start and end frequencies for each chirp.
    """
    t = np.arange(N) / fs

    a = np.random.normal(0, 10, K)
    f = np.random.uniform(0, fs/2, 2*K)
    
    x = 0
    for k in range(K):
        x += a[k] * chirp(t, f0=f[2*k], f1=f[2*k+1], t1=t[-1], method='linear')

    noise = np.random.normal(0, 5, size=N)

    # x = x + noise

    if plot:
        plt.figure(figsize=(15,3))
        plt.plot(t, x.real, label="Real part")
        plt.legend()
        plt.title(f"Dynamic signal (N={N}, order={K})")
        plt.xlabel("Time [s]")
        plt.tight_layout()

    return x, fs, f

def get_kaiser_shape(A):
    """
    Calculate the Kaiser window shape parameter (beta) based on the desired stopband attenuation.

    Parameters
    ----------
    A : float
        Desired stopband attenuation in dB.

    Returns
    -------
    beta : float
        Kaiser window shape parameter.
    """
    if A < 13.36:
        beta = 0
    elif 13.26 <= A < 60:
        beta = 0.76609 * (A - 13.26)**0.4 + 0.09834 * (A - 13.26)
    elif A >= 60:
        beta = 0.12438 * (A + 6.3)
    
    # print(f'Kaiser window shape: {beta}')
    return beta

def get_kaiser_length(Dfw, A, fs):
    """
    Calculate the required length of the Kaiser window.

    Parameters
    ----------
    Dfw : float
        Transition width in Hz.
    A : float
        Desired stopband attenuation in dB.
    fs : float
        Sampling frequency in Hz.

    Returns
    -------
    N : int
        Kaiser window length.
    """
    N = np.ceil((fs * 6 *(A + 12) / (Dfw * 155)) + 1).astype(int)
    # print(f'Kaiser window length: {N}')
    return N

def stft(x, fs=1, win=None, noverlap=0):
    """
    Short-Time Fourier Transform (STFT) using a specified window.

    Parameters
    ----------
    x : array_like
        Input signal.
    fs : float, optional
        Sampling frequency. Default is 1.
    win : array_like
        Window function to apply to each frame (must be provided).
    noverlap : int, optional
        Number of points to overlap between frames. Default is 0.

    Returns
    -------
    stft_matrix : np.ndarray
        STFT matrix (frequency bins x time frames).
    f : np.ndarray
        Frequency vector (normalized frequencies).
    t : np.ndarray
        Time vector (seconds).
    """
    assert win is not None, "Window function must be provided"
    x = np.asarray(x)
    win = np.asarray(win)
    win_len = len(win)
    hop = win_len - noverlap
    n_frames = 1 + (len(x) - win_len) // hop
    nfft = win_len
    stft_matrix = np.zeros((nfft, n_frames), dtype=np.complex64)

    for i in range(n_frames):
        start = i * hop
        frame = x[start:start + win_len] * win
        stft_matrix[:, i] = np.fft.fft(frame)

    t = np.arange(n_frames) * hop / fs
    f = np.fft.fftfreq(nfft)
    return stft_matrix, f, t


def compute_normalisation_factor(nfft, fs, win=None, mode="psd"):
    """
    Compute the normalization factor for power spectral density (PSD) or power spectrum (PS).

    Parameters
    ----------
    nfft : int
        Number of FFT points.
    fs : float
        Sampling frequency.
    win : array_like, optional
        Window function. If None, uses a rectangular window.
    mode : str, optional
        Normalization mode: "psd" for power spectral density, "ps" for power spectrum.

    Returns
    -------
    norm_factor : float
        Normalization factor for the chosen mode.
    """
    if win is None:
        win = np.ones(nfft)

    win_energy = np.sum(win ** 2)

    if mode == "psd":
        return fs * win_energy
    elif mode == "ps":
        return win_energy
    else:
        raise ValueError("mode must be either 'psd' or 'ps'")

def periodogram(x, fs, plot=False):
    """
    Estimate the Power Spectral Density (PSD) using the periodogram method.

    Parameters
    ----------
    x : array_like
        Input signal (1D array).
    fs : float
        Sampling frequency in Hz.
    plot : bool, optional
        If True, plot the estimated PSD (default is False).

    Returns
    -------
    Pxx : ndarray
        Estimated PSD values.
    """
    N = len(x)
    freqs = np.linspace(-fs/2, fs/2, N, endpoint=False)
    t = np.arange(N)/fs

    # DFT manually 
    X = np.zeros(N, dtype=complex)
    for k, f in enumerate(freqs):
        X[k] = np.sum(x * np.exp(-1j * 2 * np.pi * f * t))

    # Periodogram
    Pxx = (1/N) * np.abs(X)**2

    if plot:
        plt.figure(figsize=(7,1.5))
        plt.plot(freqs, Pxx.real, label="real")
        plt.plot(freqs, Pxx.imag, label="imaginary")
        plt.xlabel("Frequency")
        plt.ylabel("PSD")
        plt.title(f"Periodogram, N: {N}")
        plt.legend()
        plt.grid(True)
    return Pxx


def welch(x, fs, L=10, overlap=0.5, window=None, plot=False):
    """
    Estimate the Power Spectral Density (PSD) using the Welch method.

    Parameters
    ----------
    x : array_like
        Input signal (1D array).
    fs : float
        Sampling frequency in Hz.
    L : int, optional
        Number of segments to divide the signal into (default is 10).
    overlap : float, optional
        Fractional overlap between segments (0 <= overlap < 1, default is 0.5).
    window : ndarray or None, optional
        Window function to apply to each segment (default is Hann window).
    plot : bool, optional
        If True, plot the estimated PSD (default is False).

    Returns
    -------
    pxx : ndarray
        Estimated PSD values.
    freqs : ndarray
        Frequency bins corresponding to the PSD values.
    """
    N = len(x)

    if window is None:
        M = N // L
        window = np.hanning(M)
    elif isinstance(window, np.ndarray):
        M = len(window)
        print(f'Custom window set. Ignoring number of segments {L} and setting {N//M}')
        L = N // M 
    else:
        raise ValueError("window must be None or a numpy array")

    step = int(M * (1 - overlap))
    if step < 1:
        raise ValueError("Overlap too high, step size < 1")
    segments = []
    for start in range(0, N - M + 1, step):
        segments.append(x[start:start+M])
    segments = np.array(segments)
    
    pxx = 0
    for seg in segments:
        seg_win = seg * window
        pxx += periodogram(seg_win, fs)
    pxx /= len(segments)
    freqs = np.linspace(-fs/2, fs/2, M, endpoint=False)
    if plot:
        plt.plot(freqs, pxx, linestyle='--', label=f"Welch PSD | L={L}")
        plt.xlabel("Frequency [Hz]")
        plt.ylabel("PSD")
        plt.legend()
        plt.grid(True)
    return pxx, freqs

def compute_theta(rpm, fs):
    """
    Compute cumulative shaft rotation in revolutions at each sample.

    Parameters
    ----------
    rpm : scalar or array-like
        RPM values (scalar for constant rpm, or array of length N for time-varying rpm).
    fs : float
        Sampling frequency in Hz.

    Returns
    -------
    theta : float or np.ndarray
        Cumulative shaft rotation in revolutions for each sample (length N if rpm is array).
    """
    if np.isscalar(rpm):
        # constant rpm
        omega_rev_per_s = rpm / 60.0
        return omega_rev_per_s
    else:
        revs_per_s = np.asarray(rpm) / 60.0
        theta = np.cumsum(revs_per_s) / fs
        return theta

def generate_order_rpm(N, fs, rpm_start, rpm_end, ord_true):
    """
    Generate a synthetic order-tracked signal with linearly varying RPM.

    Parameters
    ----------
    N : int
        Number of samples.
    fs : float
        Sampling frequency in Hz.
    rpm_start : float
        Starting RPM value.
    rpm_end : float
        Ending RPM value.
    ord_true : float
        True order to synthesize.

    Returns
    -------
    x : np.ndarray
        Generated order-tracked signal (length N).
    fs : float
        Sampling frequency (echoed).
    rpm : np.ndarray
        Instantaneous RPM values (length N).
    """
    t = np.arange(N) / fs
    rpm = rpm_start + (rpm_end - rpm_start) * (t / t.max())

    theta = compute_theta(rpm, fs)
    if np.isscalar(theta):
        theta = (np.arange(N) * (theta / fs))

    x = 0.8 * np.sin(2*np.pi*ord_true*theta)

    return x, fs, rpm

def vsdft_frame(x_frame, theta_frame, orders, window=None):
    """
    Compute the Velocity Synchronous Discrete Fourier Transform (VSDFT) for a single frame.

    Parameters
    ----------
    x_frame : array-like
        1D array of signal samples in the frame.
    theta_frame : array-like
        1D array of cumulative shaft revolutions for each sample (same length as x_frame).
    orders : array-like
        1D array of orders to compute (can be floats).
    window : array-like or None, optional
        Window array (same length as x_frame) or None for no window.

    Returns
    -------
    S : np.ndarray
        Complex spectrum of shape (len(orders),).
    """
    x = np.asarray(x_frame)
    if window is not None:
        w = np.asarray(window)
        if w.shape != x.shape:
            raise ValueError("window must match x_frame length")
        x = x * w
    orders = np.asarray(orders)
    phase = np.exp(-1j * 2.0 * np.pi * np.outer(orders, theta_frame))
    S = phase.dot(x)
    return S


def vsdft(x, fs, orders, rpm, win, overlap=0.5):
    """
    Compute the short-time Velocity Synchronous DFT (VSDFT) order spectrogram.

    Parameters
    ----------
    x : array-like
        1D input signal (length N).
    fs : float
        Sampling frequency in Hz.
    orders : array-like
        Orders to evaluate (e.g., np.arange(1, 20)).
    rpm : float or array-like
        Scalar or 1D array of instantaneous rpm sampled at fs (length N).
    win : array-like
        Window array (length of each frame).
    overlap : float, optional
        Fractional overlap between frames [0, <1]. Default is 0.5.

    Returns
    -------
    S : np.ndarray
        Complex spectrogram (shape: [len(orders), n_frames]).
    times : np.ndarray
        Center times of frames in seconds (shape: n_frames).
    orders : np.ndarray
        Array of requested orders (shape: len(orders)).
    """
    N = len(x)
    M = len(win)
    if M < 1:
        raise ValueError("frame_len too small")
    step = int(np.round(M * (1.0 - overlap)))
    if step < 1:
        raise ValueError("overlap too large -> step < 1")

    if np.isscalar(rpm):
        revs_per_s = rpm / 60.0
        theta_full = (np.arange(N) * (revs_per_s / fs))
    else:
        rpm = np.asarray(rpm)
        if len(rpm) != N:
            raise ValueError("rpm array must have same length as x")
        theta_full = compute_theta(rpm, fs)

    frames = np.arange(0, N - M + 1, step)
    times = (frames + M/2.0) / fs

    orders = np.asarray(orders)
    S = np.zeros((len(orders), len(frames)), dtype=complex)

    for i, start in enumerate(frames):
        end = start + M
        x_frame = x[start:end]
        theta_frame = theta_full[start:end]
        S[:, i] = vsdft_frame(x_frame, theta_frame, orders, window=win)

    S /= np.sum(win)

    return S, times, orders
